#include <string>
#include <vector>
#include <cstdlib>

using namespace std;

int solution(vector<vector<int>> dots) {
    int answer = 0;
    int x1, y1, x2, y2 = 0;
    
    x1 = dots[0][0];
    y1 = dots[0][1];
    
    for(int i = 1; i < dots.size(); i++){
        if(x1 != dots[i][0] && y1 != dots[i][1]){
            x2 = dots[i][0];
            y2 = dots[i][1];
            break;
        }
    }
    
    return abs(x1-x2) * abs(y1-y2);
}
